      <textarea id="publicKey" placeholder="Paste Public Key here"></textarea>

    <!-- Private Key Input -->

    <div>

      <label for="privateKey">Private Key (for decryption):</label>

      <textarea id="privateKey" placeholder="Paste Private Key here"></textarea>

    </div>


    <!-- Message Input -->

    <div>

      <label for="messageInput">Message to Encrypt:</label>

      <textarea id="messageInput" placeholder="Enter your message to encrypt"></textarea>

    </div>


    <!-- Encrypt and Decrypt Buttons -->

    <button id="encryptButton">Encrypt</button>

    <button id="decryptButton">Decrypt</button>


    <!-- Output -->

    <div>

      <label for="encryptedOutput">Encrypted Message:</label>

      <div class="output" id="encryptedOutput"></div>

    </div>


    <div>

      <label for="decryptedOutput">Decrypted Message:</label>

      <div class="output" id="decryptedOutput"></div>

    </div>

  <script src="https://cdnjs.cloudflare.com/ajax/libs/jsencrypt/3.0.0/jsencrypt.min.js"></script>

  <script>

    // Initialize the encryption/decryption system

    const encryptButton = document.getElementById('encryptButton');

    const decryptButton = document.getElementById('decryptButton');

    const publicKeyInput = document.getElementById('publicKey');

    const privateKeyInput = document.getElementById('privateKey');

    const messageInput = document.getElementById('messageInput');

    const encryptedOutput = document.getElementById('encryptedOutput');

    const decryptedOutput = document.getElementById('decryptedOutput');


    const encrypt = () => {

      const encryptor = new JSEncrypt();

      encryptor.setPublicKey(publicKeyInput.value);


      const encrypted = encryptor.encrypt(messageInput.value);

      encryptedOutput.textContent = encrypted || 'Encryption failed!';

    };


    const decrypt = () => {

      const decryptor = new JSEncrypt();

      decryptor.setPrivateKey(privateKeyInput.value);


      const decrypted = decryptor.decrypt(encryptedOutput.textContent);

      decryptedOutput.textContent = decrypted || 'Decryption failed!';

    };


    // Event Listeners

    encryptButton.addEventListener('click', encrypt);

    decryptButton.addEventListener('click', decrypt);

  </script>
